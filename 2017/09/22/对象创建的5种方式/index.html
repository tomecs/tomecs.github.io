对象创建的5种方式

Veröffentlicht am 2017-09-22

父类

public class Father {
    public Father() {
        System.out.println("father");
    }
    {
        System.out.println("father class method block! 1  ");
    }
    static {
        System.out.println("father class static method block!");
    }
    {
        System.out.println("father class method block! 2    ");
    }
    public static void test() {
        System.out.println("Father!");
    }
}

子类

public class Son extends Father implements Cloneable, Serializable{
    {
        System.out.println("son class method block! 1  ");
    }
    static {
        System.out.println("son class static method block!");
    }
    {
        System.out.println("son class method block! 2    ");
    }
    public Son() {
        super();
        System.out.println("Son!");
    }

    public static void test() {
        System.out.println("Son!");
    }

    public Object clone() {
        Object object = null;
        try {
            object = super.clone();
        } catch (CloneNotSupportedException e) {
            e.printStackTrace();
        }
        return object;

    }
}

实验

public class Test {
    public static void main(String[] args) throws ClassNotFoundException, IllegalAccessException, InstantiationException, NoSuchMethodException, InvocationTargetException, IOException {
      //  Father father = new Father();
//        Son son = new Son();
        //第二种:使用Class类的newInstance方法 调用的是无参的构造函数
       // Son son = (Son)Class.forName("com.feng.Son").newInstance();
        //Son son1 = Son.class.newInstance();//结果与上面是相同的
        //第三种：使用Constructor类的newInstance方法
        //此方法和Class类的newInstace方法很像，但是此方法可以调用有参的和私有的构造函数
        Constructor<Son> constructor = Son.class.getConstructor();
        Son son2 = constructor.newInstance();
        //第四种：使用clone方法
        //当我们调用一个对象的clone方法，jvm就会创建一个新的对象，将前面对象的内容全部拷贝进去。
        //使用clone方法创建对象并不会调用构造函数。
        //要想使用clone方法，先实现Cloneable接口并实现其定义的clone方法
        //Son son3 = (Son)son2.clone();
        //第五种：使用反序列化
        //当我们序列化和反序列化一个对象，jvm会给我们创建一个单独的对象。
        //在反序列化时，jvm创建对象并不会调用任何构造函数。
        //为了反序化一个对象，要让类实现Senalizable接口.
        //如果父类未实现序列化接口，则反序列化出的对象会再次调用父类的构造函数来完成属于父类的那部分内容的初始化。
        ObjectOutputStream out = new ObjectOutputStream(new FileOutputStream("data.obj"));
        out.writeObject(son2);
        out.close();
        ObjectInputStream in = new ObjectInputStream(new FileInputStream("data.obj"));
        Son son3 = (Son)in.readObject();
        in.close();
    }
}
